\begin{frame}{研究背景：CoT 推理与增强策略}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{Chain-of-Thought 核心思想}
        \begin{itemize}
          \tightitems
          \item LLM 面对多步推理时直接输出答案，错误率高。
          \item CoT 通过引导模型"逐步思考"将复杂问题拆解为可解释的中间步骤。

          \item RAG + CoT（Trivedi et al., 2023）

\section{Base CoT}

\begin{frame}[fragile]{Base CoT：Chain-of-Thought 推理基础}
  \framesubtitle{Chain-of-Thought Prompting Elicits Reasoning in Large Language Models}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      \begin{block}{核心公式与流程}
        \begin{itemize}
          \tightitems
          \item Prompt 构造：将问题 $q$、选项 $O$ 与引导语拼接
          \[
            P_{\text{CoT}} = T_{\text{base}} \oplus q \oplus O \oplus \text{"Let's think step by step."}
          \]
          \item 模型生成输出后，通过答案提取函数 $f_{\text{extract}}(y)$ 得到最终预测 $\hat{a}$。
          \item 温度系数 0.7，max\_tokens=1024。
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{Base CoT 伪代码}
        \scriptsize
\begin{verbatim}
P = T.format(q, O)
y = M.generate(P, temp=0.7,
               max_tokens=1024)
a_hat = extract_answer(y)
return a_hat
\end{verbatim}
      \end{block}
      \begin{alertblock}{基线性能}
        \begin{itemize}
          \tightitems
          \item 100 样本准确率 \textbf{91\%}，单路径成本最低。
          \item 平均输出 token 187.6，是效率的参考基线。
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\subsection{Self-Consistency}

\begin{frame}{Self-Consistency 算法示意图}
  \framesubtitle{Self-Consistency Improves Chain of Thought Reasoning in Language Models}
  \centering
  \includegraphics[height=0.78\textheight,keepaspectratio]{self-consistency.png}
\end{frame}

\begin{frame}{Self-Consistency：从单条推理变成多路径投票}
  \framesubtitle{Self-Consistency Improves Chain of Thought Reasoning in Language Models}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.49\textwidth}
      \begin{block}{论文核心思想}
        \begin{itemize}
          \tightitems
          \item 用较高温度采样多条 CoT 推理路径。
          \item 每条路径给出一个最终选项。
          \item 通过聚合答案降低单次采样偶然错误。
        \end{itemize}
      \end{block}
      \begin{block}{原始多数投票}
        \centering
        路径数多的答案获胜：\quad $\arg\max_y \sum_i \mathbbm{1}[\hat y_i=y]$
      \end{block}
    \end{column}
    \begin{column}{0.47\textwidth}
      \begin{alertblock}{项目中的改进点}
        多数票不一定代表高质量推理。因此我们负责的 self-consistency 部分改为：
        \[
          \arg\max_y \sum_i q_i \cdot \mathbbm{1}[\hat y_i=y]
        \]
        其中 $q_i$ 是第 $i$ 条路径的启发式质量分。
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{自适应加权 Self-Consistency：票数之外再看路径质量}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.54\textwidth}
      \begin{block}{质量分由什么决定}
        \begin{tabularx}{\linewidth}{lY}
          \toprule
          信号 & 作用 \\
          \midrule
          答案清晰 & 是否严格输出 \texttt{Answer: X} \\
          推理步骤 & 是否有显式 step 或分段推理 \\
          数学痕迹 & 是否包含数字、运算符、等式等计算信号 \\
          输出长度 & 过短扣分，过长轻微扣分 \\
          自相矛盾 & 多处答案不一致时扣分 \\
          \bottomrule
        \end{tabularx}
      \end{block}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{block}{一个直观例子}
        \small
        \begin{tabular}{lcc}
          \toprule
          答案 & 路径数 & 加权票 \\
          \midrule
          A & 2 & $1.8 + 1.7 = 3.5$ \\
          B & 3 & $0.8 + 0.7 + 0.6 = 2.1$ \\
          \bottomrule
        \end{tabular}
        \smallskip

        旧算法选 B，因为 B 有 3 票；新算法选 A，因为 A 的两条推理更清晰、更完整。
      \end{block}
      \begin{alertblock}{边界}
        该方法只使用本路径文本的局部信号，不调用 verifier、不检索外部知识，也不引入多 Agent。
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{结合真实代码：self\_consistency.py 的执行流程}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.49\textwidth}
      \begin{block}{关键函数}
        \begin{itemize}
          \tightitems
          \item \texttt{\_path\_quality}: 计算每条路径的启发式质量分。
          \item \texttt{\_aggregate\_votes}: 按权重聚合，并用原始票数、最佳路径质量、出现顺序打破平票。
          \item \texttt{\_can\_stop\_early}: 当领先答案无法被剩余路径追上时提前停止。
          \item \texttt{run}: 生成路径、抽取答案、重试空答案、聚合最终预测。
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.47\textwidth}
      \begin{block}{伪代码}
        \scriptsize
\begin{verbatim}
for i in range(n_paths):
    raw = model.generate(prompt, temperature)
    pred = task.extract_answer(raw)
    if retry_on_empty and not pred:
        raw = model.generate(prompt, low_temp)
    weight = _path_quality(raw, pred)
    save(pred, weight)
    if _can_stop_early(...):
        break

final = _aggregate_votes(predictions, weights)
\end{verbatim}
      \end{block}
      \smallnote{低温重试让模型输出更稳定，主要用于补救没有 \texttt{Answer: X} 的样本。}
    \end{column}
  \end{columns}
\end{frame}

\subsection{Prefix Consistency}

\begin{frame}{Prefix Consistency 算法示意图}
  \framesubtitle{Reliable Chain-of-Thought via Prefix Consistency}
  \centering
  \includegraphics[height=0.78\textheight,width=\textwidth,keepaspectratio]{prefix-consistency.png}
\end{frame}

\begin{frame}{Prefix Consistency：截断-再生评估推理链可靠性}
  \framesubtitle{Reliable Chain-of-Thought via Prefix Consistency (Iwase et al., 2026)}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      \begin{block}{核心机制}
        \begin{enumerate}
          \tightitems
          \item 生成 $N=3$ 条初始 CoT 路径。
          \item 在每条路径中间位置 $t_i = \lfloor |y_i| \cdot r \rfloor$ 截断（$r=0.5$），得到前缀 $p_i$。
          \item 从 $p_i$ 再生 $K=3$ 次，得到补全路径。
          \item 计算前缀一致性得分：
          \[
            C_i = \frac{1}{K} \sum_{j=1}^{K} \mathbbm{1}[\text{extract\_answer}(y_{i,j}^{\prime}) = a_i]
          \]
        \end{enumerate}
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{加权投票}
        \[
          \hat{a} = \arg\max_{a} \sum_{i=1}^{N} W(C_i) \cdot \mathbbm{1}[a_i = a]
        \]
        \vspace{-0.3em}
        权重函数可选：linear / quadratic / cubic / unanimous
      \end{block}
      \begin{alertblock}{实验亮点}
        \begin{itemize}
          \tightitems
          \item 100 样本准确率 \textbf{93.0\%}。
          \item 平均输出 token 仅 \textbf{159.9}（0.85× base CoT），token 效率最优。
          \item 无需额外模型，纯 prompt 工程方法。
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\subsection{Step-Aware Verifier}

\begin{frame}{Step-Aware Verifier 算法示意图}
  \framesubtitle{Making Large Language Models Better Reasoners with Step-Aware Verifier}
  \centering
  \includegraphics[height=0.78\textheight,keepaspectratio]{verifier.png}
\end{frame}

\begin{frame}{LLM Verifier（原版）：每一步都要调一次 API}
  \framesubtitle{Making Large Language Models Better Reasoners with Step-Aware Verifier}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      \begin{block}{论文思路}
        \begin{itemize}
          \tightitems
          \item 生成多条推理路径。
          \item 每条路径拆成 step，每一步调用 LLM 打分（Score: 1-10）。
          \item 选平均分最高的路径作为最终答案。
        \end{itemize}
      \end{block}
      \begin{block}{实现代码}
        \small \texttt{strategies/step\_verifier.py} 中 \texttt{\_verify\_step()} 对每个 step
        构造 \texttt{prompts/step\_verifier.txt} 模板，调用 \texttt{model.generate()} 打分。
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{alertblock}{性能代价}
        \begin{itemize}
          \tightitems
          \item 每条路径每步 = 1 次 API 调用。
          \item 10 step $\times$ 5 路径 = 50 次 API / 题。
          \item 50 样本实测: \textbf{206.6s/条}，极慢。
        \end{itemize}
      \end{alertblock}
      \begin{block}{聚合}
        对每条路径求 step 分数均值，取最高值：
        \[
          \hat{a} = \arg\max_{i} \frac{1}{|y_i|} \sum_{s \in y_i} V(s)
        \]
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{本地 DeBERTa Verifier：纯 CUDA 推理，零额外 API}
  \framesubtitle{自定义微调 DeBERTa-v3-large 替代 LLM 验证}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      \begin{block}{模型架构与评分}
        \footnotesize
        \begin{itemize}
          \tightitems
          \item Token 分类训练，推理时\textbf{当序列分类用}——只看 [CLS] 的 \texttt{SOLUTION-CORRECT} 概率。
          \item STEP 标签仅辅助训练（loss 0.1），推理不用。

        \footnotesize
        \texttt{--n\_prompts}：每道题用多个 random few-shot prompt。
        \[
          \hat{a} = \arg\max_{a} \sum_{i=1}^{N} V(y_i) \cdot \mathbb{1}[a_i = a]
        \]
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{关键文件}
        \footnotesize
        \begin{tabularx}{\linewidth}{lY}
          \midrule
          \texttt{step\_verifier.py} & 策略主逻辑 \\

        \textbf{验证纯 CUDA，零 API 调用。}\\
        50 样本: \textbf{52.3s/条}（LLM 版 $\frac{1}{4}$）。

\subsection{RAG + CoT}

\begin{frame}{RAG + CoT 算法示意图}
  \framesubtitle{Interleaving Retrieval with Chain-of-Thought Reasoning}
  \centering
  \includegraphics[height=0.78\textheight,keepaspectratio]{rag.png}
\end{frame}

\begin{frame}{RAG + CoT：检索增强的两阶段推理}
  \framesubtitle{Interleaving Retrieval with Chain-of-Thought Reasoning for Knowledge-Intensive Multi-Step Questions}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      \begin{block}{两阶段流程}
        \textbf{Phase 1 --- 检索：}
        $\mathcal{C}_q = \text{retrieve}(q, k)$，从知识库召回 top-$k$ 文档（$k=3$）。
        \textbf{Phase 2 --- 推理：}
        将上下文拼入 prompt，生成推理链：
        \[
          P_{\text{RAG}} = T_{\text{rag}}(\text{format}(\mathcal{C}_q), q, O)
        \]
      \end{block}
      \begin{block}{Jaccard 关键词检索器}
        \begin{itemize}
          \tightitems
          \item 分词：正则 \texttt{[a-z]+}，过滤约 100 个停用词，保留 $|t|>2$。
          \item 相似度：$\text{sim}_{\text{Jaccard}} = \frac{|A \cap B|}{|A \cup B|}$，截断 top-$k$。
          \item 空结果降级为 base CoT。
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{知识库}
        \begin{itemize}
          \tightitems
          \item 15 条手工整理的通用数学事实。
          \item 分类：math\_basics / geometry / probability。
          \item 可热插拔，运行时自动加载。
        \end{itemize}
      \end{block}
      \begin{alertblock}{实验观察与局限}
        \begin{itemize}
          \tightitems
          \item 100 样本准确率 \textbf{92\%}，高于 base（91\%）。
          \item Jaccard 缺乏语义理解，知识库仅 15 条是主要瓶颈。
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\subsection{Multi-Agent Debate}

\begin{frame}{Multi-Agent Debate：5 角色协作与交叉评审}
  \framesubtitle{Improving Factuality and Reasoning in Language Models through Multiagent Debate}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      \begin{block}{5 种认知角色}
        \begin{tabular}{ll}
          \toprule
          角色 & 职责 \\
          \midrule
          分析师 (T=0.3) & 逐步推导、代入验证 \\
          批判者 (T=0.8) & 质疑思路、寻找陷阱 \\
          直觉者 (T=1.2) & 模式识别、联想题型 \\
          验证者 (T=0.5) & 代入数值、边界检查 \\
          综合者 (T=0.7) & 整合各方、权衡决策 \\
          \bottomrule
        \end{tabular}
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{核心流程（3 轮）}
        \begin{enumerate}
          \tightitems
          \item 5 Agent 并行独立回答。
          \item 交叉评审：各 Agent 审阅他人答案，指出漏洞。
          \item 修订：基于辩论记录修正答案。
          \item 收敛检测：连续两轮一致则提前停止。
          \item 多数投票确定最终答案。
        \end{enumerate}
      \end{block}
      \begin{alertblock}{实验亮点}
        \begin{itemize}

\begin{frame}{各方法优越性与局限性}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{主要优越性}
        \begin{itemize}
          \tightitems
          \item \textbf{Self-Consistency：} 94\% 稳定，路径质量评分 + 早停节省 $\sim$30\% API 调用。
          \item \textbf{Prefix Consistency：} token 效率最优（0.85× base），无需额外模型，纯 prompt 工程。
          \item \textbf{DeBERTa Verifier：} 零 API 费用，速度比 LLM 版快 4×（52.3s vs 206.6s/条）。
          \item \textbf{Multi-Agent Debate：} 最高准确率 95\%，5 种认知风格 + 交叉评审。
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{主要局限性}
        \begin{itemize}
          \tightitems
          \item \textbf{Debate：} 实际成本 5$\sim$7× base CoT，温度差异大时可能无法收敛。
          \item \textbf{RAG + CoT：} Jaccard 缺乏语义理解，知识库仅 15 条。
          \item \textbf{Step Verifier (LLM)：} 速度极慢（206.6s/条），token 消耗 3× base。
          \item \textbf{Prefix Consistency：} 墙钟时间较长（64.4s/条），截断比固定 0.5。
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{主要结论}
  \begin{enumerate}
    \tightitems
    \item \textbf{Multi-Agent Debate 准确率最高（95.0\%）：} 5 Agent 并行 + 交叉评审 + 收敛检测，是追求极致准确率的上限方案。
    \item \textbf{Self-Consistency 最鲁棒（94.0\% 稳定）：} 50 和 100 样本均保持 94\%，路径质量评分 + 早停实现高 token 效率。
    \item \textbf{Prefix Consistency token 效率最优：} 平均输出 token 仅 159.9（0.85× base），准确率 93.0\%，验证了截断-再生一致性信号的可靠性。
    \item \textbf{高质量 Feedback 比复杂架构更关键：} 覆盖数 3/5 的 Self-Consistency 超越覆盖数 5/5 的 Step Verifier，证明\textbf{局部评估质量}比\textbf{系统复杂度}更重要。
    \item \textbf{RAG 受检索器质量严重制约：} 当前 Jaccard 检索器是瓶颈，升级 Embedding 语义检索后潜力大。
  \end{enumerate}

  \vspace{0.5em}
  \begin{block}{未来工作}
    \small
    \begin{itemize}
      \tightitems
      \item Debate 优化：自适应角色配置、引入仲裁者 Agent、基于置信度的加权投票。
      \item 检索器升级：BGE / E5 语义检索替代 Jaccard，扩充知识库规模。
      \item 扩展验证：在 GSM8K、MATH 等更大数据集上验证泛化能力。
      \item Tree-of-Thought：将多路径策略扩展为树状搜索结构。
    \end{itemize}
  \end{block}
\end{frame}
